 var input = Console.ReadLine().Split().Select(int.Parse).ToArray();
            int currentPosition = int.Parse(Console.ReadLine());
            int currentDamage = 1;
            while (true)
            {
                var directionSteps = Console.ReadLine().Split().ToArray();
                var direction = directionSteps[0];
                if (directionSteps[0] == "Supernova")
                {
                    break;
                }
                int steps = int.Parse(directionSteps[1]);
                
                if (direction == "right")
                {
                    for (int i = 0; i < steps; i++)
                    {
                        currentPosition += 1;
                        if (currentPosition > input.Length)
                        {
                            currentPosition = 0;
                            currentDamage++;
                        }
                        input[currentPosition] -= currentDamage;
                    }
                }
                if (direction == "left")
                {
                    for (int i = 0; i < steps; i++)
                    {
                        currentPosition -= 1;
                        if (currentPosition < 0)
                        {
                            currentPosition = input.Length - 1;
                            currentDamage++;
                        }
                        input[currentPosition] -= currentDamage;

                    }
                }
            }

            Console.WriteLine(string.Join(" ", input));
