 using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.Threading.Tasks;

namespace icarus_shiban
{
    class Program
    {
        static void Main(string[] args)
        {
            var input = Console.ReadLine().Split().Select(decimal.Parse).ToArray();
            var position = decimal.Parse(Console.ReadLine());
            decimal damage=1;

            while (true)
            {
                var input2 = Console.ReadLine().Split().ToArray();
                if (input2[0] == "Supernova")
                {
                    Console.WriteLine(string.Join(" ", input));
                    break;
                }

                if (input2[0] == "right")
                {
                    for (int i = 0; i < decimal.Parse(input2[1]); i++)
                    {

                        position += 1;
                        if (position == input.Length + 1)
                        {
                            position = input[0];
                            damage++;
                        }
                        input[(long)position] -= damage;
                    }
                }

                if (input2[0] == "left")
                {
                    for (long i = 0; i < long.Parse(input2[1]); i++)
                    {

                        position -= 1;
                        if (position == -1)
                        {
                            position = input.Length-1;
                            damage++;
                        }
                        if (input[(int)position] >0)
                        {

                            input[(long)position] -= damage;
                        }
                    }
                }
            }

            
        }
    }
}
